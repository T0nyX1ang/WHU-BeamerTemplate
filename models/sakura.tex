\providecommand{\drawsakura}[7]{
    % draw a sakura
    % required parameter(s): #1 -- outer color of the sakura
    %                        #2 -- inner color of the sakura
    %                        #3 -- kernel color of the sakura    
    %                        #4 -- outwards angle of the sakura
    %                        #5 -- inwards angle of the sakura
    %                        #6 -- offset in x axis
    %                        #7 -- offset in y axis
	\input{./models/petal.tex}
    \foreach \x in {1,...,5}			
	{
        \drawpetal{-(\x - 1) * 72}{#1}{#2}{#3}{#4}{#5}{#6}{#7}
	}
}
